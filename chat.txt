# Ejemplo en Python (usando Pandas para análisis)
import pandas as pd
import re

# Cargar chat
chat_data = pd.read_csv("chat_whatsapp.txt", sep="]", names=["fecha", "mensaje"])

# Extraer usuario y mensaje
chat_data["usuario"] = chat_data["mensaje"].apply(lambda x: x.split(":")[0].strip())
chat_data["texto"] = chat_data["mensaje"].apply(lambda x: x.split(":")[1].strip() if ":" in x else "")

# Función para "¿Quién es más probable que diga 'jaja'?"
def quien_dice_mas(palabra):
    conteo = {}
    for usuario in chat_data["usuario"].unique():
        mensajes_usuario = chat_data[chat_data["usuario"] == usuario]["texto"]
        apariciones = sum(mensajes_usuario.str.count(re.escape(palabra)))
        conteo[usuario] = apariciones
    return max(conteo, key=conteo.get)

print("¿Quién es más probable que diga 'jaja'? ->", quien_dice_mas("jaja"))
